What notification-service does

Pure Kafka consumer — no REST endpoints (except health check). Listens to Kafka topics: payment.failed, payment.success, subscription.expired. When an event arrives, it sends email/SMS to the user. Stateless — no database needed. This is the correct microservices pattern: notification logic is completely decoupled from business logic.

Without Kafka (current state): We simulate by exposing a POST /notifications/send endpoint that subscription-service can call directly. Kafka is added later.

5NotificationEvent.java — shared event model
NotificationEvent.javaNEWcom.ott.notification.event
package com.ott.notification.event;
import java.math.BigDecimal;
import java.time.LocalDateTime;

/**
 * Common event model. Kafka messages are deserialized into this class.
 * payment-service and subscription-service serialize their events
 * using the same structure.
 *
 * eventType values:
 *   PAYMENT_SUCCESS      → "Your payment was successful"
 *   PAYMENT_FAILED       → "Your payment failed, please retry"
 *   SUBSCRIPTION_EXPIRED → "Your subscription has expired"
 *   SUBSCRIPTION_EXPIRING→ "Your subscription expires in 3 days"
 *   SUBSCRIPTION_CREATED → "Welcome! Your subscription is active"
 */
public class NotificationEvent {
    private String eventType;
    private Long userId;
    private String username;
    private String email;
    private String planId;
    private String planName;
    private BigDecimal amount;
    private String failureReason;
    private LocalDateTime eventTime;

    public NotificationEvent() {}

    public String getEventType() { return eventType; }
    public void setEventType(String v) { eventType = v; }
    public Long getUserId() { return userId; }
    public void setUserId(Long v) { userId = v; }
    public String getUsername() { return username; }
    public void setUsername(String v) { username = v; }
    public String getEmail() { return email; }
    public void setEmail(String v) { email = v; }
    public String getPlanId() { return planId; }
    public void setPlanId(String v) { planId = v; }
    public String getPlanName() { return planName; }
    public void setPlanName(String v) { planName = v; }
    public BigDecimal getAmount() { return amount; }
    public void setAmount(BigDecimal v) { amount = v; }
    public String getFailureReason() { return failureReason; }
    public void setFailureReason(String v) { failureReason = v; }
    public LocalDateTime getEventTime() { return eventTime; }
    public void setEventTime(LocalDateTime v) { eventTime = v; }
}

7KafkaConsumerService.java — listens to all topics
KafkaConsumerService.javaNEWcom.ott.notification.kafka
package com.ott.notification.kafka;
import com.ott.notification.event.NotificationEvent;
import com.ott.notification.service.EmailNotificationService;
import org.slf4j.*;
import org.springframework.kafka.annotation.KafkaListener;
import org.springframework.stereotype.Service;

/**
 * PURPOSE: Listens to Kafka topics and triggers email notifications.
 *
 * WHY @KafkaListener:
 * Each method with @KafkaListener is automatically called by Spring
 * whenever a message arrives on the specified topic.
 * No manual polling needed — Spring handles it.
 *
 * groupId = "notification-group":
 * All notification-service instances share one group.
 * Kafka ensures only ONE instance processes each message —
 * prevents duplicate emails if you scale to 2 instances.
 *
 * NOTE: These listeners are INACTIVE until:
 *   1. Kafka is running (localhost:9092)
 *   2. spring.kafka.bootstrap-servers is uncommented in application.properties
 *   3. Topics are created (auto-created on first publish if configured)
 *
 * For NOW without Kafka: use NotificationController POST endpoint to test.
 */
@Service
public class KafkaConsumerService {

    private static final Logger log = LoggerFactory.getLogger(KafkaConsumerService.class);
    private final EmailNotificationService emailService;

    public KafkaConsumerService(EmailNotificationService e) { emailService = e; }

    /** Triggered when payment.success is published by payment-service */
    @KafkaListener(topics = "payment.success", groupId = "notification-group")
    public void handlePaymentSuccess(NotificationEvent event) {
        log.info("Kafka: payment.success received for user: {}", event.getUserId());
        emailService.sendPaymentSuccessEmail(event);
    }

    /** Triggered when payment.failed is published by payment-service */
    @KafkaListener(topics = "payment.failed", groupId = "notification-group")
    public void handlePaymentFailed(NotificationEvent event) {
        log.info("Kafka: payment.failed received for user: {}", event.getUserId());
        emailService.sendPaymentFailedEmail(event);
    }

    /** Triggered when subscription.expiring is published by subscription-service scheduler */
    @KafkaListener(topics = "subscription.expiring", groupId = "notification-group")
    public void handleSubscriptionExpiring(NotificationEvent event) {
        log.info("Kafka: subscription.expiring for user: {}", event.getUserId());
        emailService.sendSubscriptionExpiringEmail(event);
    }

    /** Triggered when subscription.expired is published by subscription-service scheduler */
    @KafkaListener(topics = "subscription.expired", groupId = "notification-group")
    public void handleSubscriptionExpired(NotificationEvent event) {
        log.info("Kafka: subscription.expired for user: {}", event.getUserId());
        emailService.sendSubscriptionExpiredEmail(event);
    }
}
8NotificationController.java — fallback REST endpoint for testing WITHOUT Kafka
package com.ott.notification.controller;
import com.ott.notification.event.NotificationEvent;
import com.ott.notification.service.EmailNotificationService;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.*;

/**
 * PURPOSE: Fallback REST endpoint for testing notifications WITHOUT Kafka.
 * In production: remove this and use only Kafka consumers.
 * For testing: POST /notifications/send with event body.
 *
 * Also used by subscription-service as fallback when Kafka is not available.
 *
 * Postman: POST http://localhost:8080/notifications/send (with Bearer token)
 * Body: {"eventType":"PAYMENT_SUCCESS","userId":1,"username":"john.doe",
 *        "email":"john@gmail.com","planName":"Disney+ Basic","amount":199.0}
 */
@RestController
@RequestMapping("/notifications")
public class NotificationController {

    private final EmailNotificationService emailService;
    public NotificationController(EmailNotificationService e) { emailService = e; }

    @PostMapping("/send")
    public ResponseEntity<String> sendNotification(@RequestBody NotificationEvent event) {
        switch (event.getEventType()) {
            case "PAYMENT_SUCCESS"    -> emailService.sendPaymentSuccessEmail(event);
            case "PAYMENT_FAILED"     -> emailService.sendPaymentFailedEmail(event);
            case "SUBSCRIPTION_EXPIRING" -> emailService.sendSubscriptionExpiringEmail(event);
            case "SUBSCRIPTION_EXPIRED"  -> emailService.sendSubscriptionExpiredEmail(event);
            default -> throw new RuntimeException("Unknown event type: " + event.getEventType());
        }
        return ResponseEntity.ok("Notification sent for event: " + event.getEventType());
    }

    @GetMapping("/health-check")
    public ResponseEntity<String> healthCheck() {
        return ResponseEntity.ok("notification-service is running");
    }
}
9GlobalExceptionHandler.java
package com.ott.notification.exception;
import jakarta.servlet.http.HttpServletRequest;
import org.springframework.http.*;
import org.springframework.web.bind.annotation.*;
import java.time.LocalDateTime;
import java.util.*;

@RestControllerAdvice
public class GlobalExceptionHandler {
    @ExceptionHandler(RuntimeException.class)
    public ResponseEntity<Map<String,Object>> handle(RuntimeException ex, HttpServletRequest req) {
        Map<String,Object> err = new LinkedHashMap<>();
        err.put("status", 400); err.put("message", ex.getMessage());
        err.put("path", req.getRequestURI()); err.put("timestamp", LocalDateTime.now());
        return ResponseEntity.status(HttpStatus.BAD_REQUEST).body(err);
    }
}

What Config Server does

Centralizes all application.properties for every service into one GitHub repository. All services read their config from Config Server on startup instead of their local properties file. Change DB password in one place → all services pick it up (some without restart).

Disney use case: When RDS endpoint changes, Config Server is updated once. All 6 services automatically get the new URL. Without Config Server: update 6 files, redeploy 6 services.

Two types of load balancing in our system

Client-side (Spring Cloud LoadBalancer): When subscription-service calls catalog-service via Feign and catalog has 3 instances, Spring automatically distributes calls round-robin across all 3.

Server-side (AWS ALB): When ESPN UI hits the API Gateway and there are 2 gateway instances, ALB distributes incoming HTTP traffic between them.

The Problem

subscription-service has a users table in subscription_db. user-service has a users table in user_db. Two tables = data duplication = validation inconsistencies.

Root cause: subscription-service was built first without user-service. It needed user records to create subscriptions, so it created its own users table temporarily.

The Solution

Database-per-service principle: user_db is the single source of truth for user data. subscription_db stores only userId (as a plain column, not FK) as a reference. All user validation happens by calling user-service via Feign.

What stays: subscriptions table keeps userId column. Users table in subscription_db can be removed.

What changes: SubscriptionService no longer looks up users in subscription_db — it calls user-service via Feign to validate user exists.

1Add UserServiceClient Feign to subscription-service
UserServiceClient.javaNEWcom.ott.subscription.client
package com.ott.subscription.client;
import org.springframework.cloud.openfeign.FeignClient;
import org.springframework.web.bind.annotation.*;
import java.util.Map;

/**
 * PURPOSE: Validates user existence by calling user-service.
 * Replaces local UserRepository in subscription-service.
 *
 * When subscription-service receives a subscribe request with userId=5,
 * instead of querying its own local users table, it calls
 * user-service to confirm user 5 exists and is active.
 *
 * This is the correct microservices approach — user-service owns user data.
 */
@FeignClient(name = "user-service", url = "${user.service.url}")
public interface UserServiceClient {

    /** Check if user exists and is active before creating subscription */
    @GetMapping("/auth/me/{userId}")
    Map<String, Object> getUserById(@PathVariable("userId") Long userId);
}

5ReportService.java — all report logic with SQL queries
package com.ott.report.service;
import org.slf4j.*;
import org.springframework.jdbc.core.JdbcTemplate;
import org.springframework.stereotype.Service;
import java.time.LocalDate;
import java.util.List;
import java.util.Map;

/**
 * PURPOSE: Queries all databases and generates operational reports.
 * Uses JdbcTemplate for direct SQL (no JPA entities needed for reports).
 *
 * DISNEY CONTEXT: These exact reports were generated at Tavant:
 * 1. Payment dues — outstanding payment failures
 * 2. Expiry alerts — subscriptions ending soon
 * 3. Daily revenue summary — total collections
 * 4. Renewal pipeline — upcoming renewals
 */
@Service
public class ReportService {

    private static final Logger log = LoggerFactory.getLogger(ReportService.class);
    private final JdbcTemplate jdbc;
    public ReportService(JdbcTemplate j) { jdbc = j; }

    /** Report 1: Users with failed payments in last 7 days */
    public void generatePaymentDuesReport() {
        String sql = """
            SELECT p.user_id, p.plan_id, p.amount, p.failure_reason, p.created_at
            FROM payment_db.payments p
            WHERE p.status = 'FAILED'
            AND p.created_at >= DATE_SUB(NOW(), INTERVAL 7 DAY)
            ORDER BY p.created_at DESC
            """;
        try {
            List<Map<String, Object>> results = jdbc.queryForList(sql);
            log.info("=== PAYMENT DUES REPORT ===");
            log.info("Failed payments in last 7 days: {}", results.size());
            results.forEach(row ->
                log.info("  User:{} Plan:{} Amount:{} Reason:{} Date:{}",
                    row.get("user_id"), row.get("plan_id"),
                    row.get("amount"), row.get("failure_reason"), row.get("created_at")));
            log.info("===========================");
        } catch (Exception e) {
            log.error("Payment dues report failed: {}", e.getMessage());
        }
    }

    /** Report 2: Subscriptions expiring in next 7 days */
    public void generateExpiryReport() {
        String sql = """
            SELECT s.subscription_id, s.user_id, s.plan_id, s.end_date, s.auto_renewal
            FROM subscription_db.subscriptions s
            WHERE s.status = 'ACTIVE'
            AND s.end_date BETWEEN CURDATE() AND DATE_ADD(CURDATE(), INTERVAL 7 DAY)
            ORDER BY s.end_date ASC
            """;
        try {
            List<Map<String, Object>> results = jdbc.queryForList(sql);
            log.info("=== SUBSCRIPTION EXPIRY REPORT (Next 7 Days) ===");
            log.info("Expiring subscriptions: {}", results.size());
            results.forEach(row ->
                log.info("  SubId:{} User:{} Plan:{} Expires:{} AutoRenew:{}",
                    row.get("subscription_id"), row.get("user_id"),
                    row.get("plan_id"), row.get("end_date"), row.get("auto_renewal")));
            log.info("================================================");
        } catch (Exception e) {
            log.error("Expiry report failed: {}", e.getMessage());
        }
    }

    /** Report 3: Today's payment summary */
    public void generateDailySummary() {
        String sql = """
            SELECT
                status,
                COUNT(*) as count,
                SUM(amount) as total_amount
            FROM payment_db.payments
            WHERE DATE(created_at) = CURDATE()
            GROUP BY status
            """;
        try {
            List<Map<String, Object>> results = jdbc.queryForList(sql);
            log.info("=== DAILY PAYMENT SUMMARY ({}) ===", LocalDate.now());
            results.forEach(row ->
                log.info("  Status:{} Count:{} Total:₹{}",
                    row.get("status"), row.get("count"), row.get("total_amount")));
            log.info("==================================");
        } catch (Exception e) {
            log.error("Daily summary failed: {}", e.getMessage());
        }
    }

    /** Report 4: Weekly subscription growth */
    public void generateWeeklyGrowthReport() {
        String newSubsSql = "SELECT COUNT(*) FROM subscription_db.subscriptions WHERE created_at >= DATE_SUB(NOW(), INTERVAL 7 DAY) AND status='ACTIVE'";
        String cancelledSql = "SELECT COUNT(*) FROM subscription_db.subscriptions WHERE updated_at >= DATE_SUB(NOW(), INTERVAL 7 DAY) AND status='CANCELLED'";
        try {
            Integer newSubs = jdbc.queryForObject(newSubsSql, Integer.class);
            Integer cancelled = jdbc.queryForObject(cancelledSql, Integer.class);
            log.info("=== WEEKLY GROWTH REPORT ===");
            log.info("  New subscriptions: {}", newSubs);
            log.info("  Cancellations:     {}", cancelled);
            log.info("  Net growth:        {}", (newSubs != null ? newSubs : 0) - (cancelled != null ? cancelled : 0));
            log.info("=============================");
        } catch (Exception e) {
            log.error("Weekly growth report failed: {}", e.getMessage());
        }
    }
}

 *
