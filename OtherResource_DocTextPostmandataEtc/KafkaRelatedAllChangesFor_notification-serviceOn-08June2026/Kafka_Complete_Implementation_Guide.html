Start Kafka locally on Windows 10 — using Docker (easiest)

Why Docker for Kafka?

Installing Kafka manually on Windows requires Java configuration, ZooKeeper setup, environment variables — takes 1-2 hours and often breaks. Docker starts the entire Kafka stack in 30 seconds with one command. You already have Docker Desktop running.

1Create docker-compose-kafka.yml — Kafka + Zookeeper

Create a new file docker-compose-kafka.yml anywhere on your Desktop or project folder. Open Notepad → paste → Save As → All Files → docker-compose-kafka.yml

Shared Event Model — used by ALL services

The problem — different services need the same event class

payment-service PRODUCES a PaymentEvent. notification-service CONSUMES that PaymentEvent. Both must use the identical class structure — same field names, same types — otherwise JSON deserialization fails.

Best practice (Production): Create a separate shared library JAR that all services import. This ensures they always use the same version.

Our approach (simpler for learning): Create the same event class in each service with identical structure. Both payment-service and notification-service have their own copy — they look the same, so JSON mapping works.

This SAME class must exist in:

  • com.ott.payment.event.NotificationEvent (in payment-service)
  • com.ott.subscription.event.NotificationEvent (in subscription-service)
  • com.ott.notification.event.NotificationEvent (already created in notification-service)

KafkaConfig.javaNEW FILEcom.ott.payment.config
package com.ott.payment.config;

import com.ott.payment.event.NotificationEvent;
import org.apache.kafka.clients.producer.ProducerConfig;
import org.apache.kafka.common.serialization.StringSerializer;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.kafka.core.*;
import org.springframework.kafka.support.serializer.JsonSerializer;
import java.util.HashMap;
import java.util.Map;

/**
 * PURPOSE: Configures Kafka producer for payment-service.
 * Creates KafkaTemplate bean that PaymentEventPublisher uses
 * to send messages to Kafka topics.
 *
 * WHY this config class?
 * Spring Boot auto-configuration handles basic Kafka setup,
 * but we need typed KafkaTemplate<String, NotificationEvent>
 * for type safety. Without this, you'd use raw Map objects.
 */
@Configuration
public class KafkaConfig {

    @Value("${spring.kafka.bootstrap-servers}")
    private String bootstrapServers;

    @Bean
    public ProducerFactory<String, NotificationEvent> producerFactory() {
        Map<String, Object> config = new HashMap<>();
        config.put(ProducerConfig.BOOTSTRAP_SERVERS_CONFIG, bootstrapServers);
        config.put(ProducerConfig.KEY_SERIALIZER_CLASS_CONFIG, StringSerializer.class);
        config.put(ProducerConfig.VALUE_SERIALIZER_CLASS_CONFIG, JsonSerializer.class);
        // Don't add type headers — simpler for cross-service compatibility
        config.put(JsonSerializer.ADD_TYPE_INFO_HEADERS, false);
        return new DefaultKafkaProducerFactory<>(config);
    }

    @Bean
    public KafkaTemplate<String, NotificationEvent> kafkaTemplate() {
        return new KafkaTemplate<>(producerFactory());
    }
}
4PaymentEventPublisher.java — NEW class that publishes to Kafka
PaymentEventPublisher.javaNEW FILEcom.ott.payment.kafka
package com.ott.payment.kafka;

import com.ott.payment.event.NotificationEvent;
import org.slf4j.*;
import org.springframework.kafka.core.KafkaTemplate;
import org.springframework.kafka.support.SendResult;
import org.springframework.stereotype.Component;
import java.util.concurrent.CompletableFuture;

/**
 * PURPOSE: Publishes payment events to Kafka topics.
 *
 * Called by: PaymentService AFTER payment is processed.
 * Publishes to: "payment.success" or "payment.failed"
 * Consumed by: notification-service (sends email)
 *              subscription-service (activates or keeps pending)
 *
 * KEY: userId is used as the Kafka message KEY.
 * WHY: Messages with same key always go to same partition.
 * This guarantees order — all events for user 5 are processed in order.
 */
@Component
public class PaymentEventPublisher {

    private static final Logger log = LoggerFactory.getLogger(PaymentEventPublisher.class);

    private static final String TOPIC_SUCCESS = "payment.success";
    private static final String TOPIC_FAILED  = "payment.failed";

    private final KafkaTemplate<String, NotificationEvent> kafkaTemplate;

    public PaymentEventPublisher(KafkaTemplate<String, NotificationEvent> kafkaTemplate) {
        this.kafkaTemplate = kafkaTemplate;
    }

    /**
     * Publish payment SUCCESS event.
     * Called after payment-service confirms payment went through.
     * notification-service will send "Welcome to Disney+!" email.
     */
    public void publishPaymentSuccess(NotificationEvent event) {
        event.setEventType("PAYMENT_SUCCESS");
        sendMessage(TOPIC_SUCCESS, event);
    }

    /**
     * Publish payment FAILED event.
     * Called after payment-service detects card declined / error.
     * notification-service will send "Payment failed, please retry" email.
     */
    public void publishPaymentFailed(NotificationEvent event) {
        event.setEventType("PAYMENT_FAILED");
        sendMessage(TOPIC_FAILED, event);
    }

    private void sendMessage(String topic, NotificationEvent event) {
        // Key = userId as String — ensures ordering per user
        String key = event.getUserId() != null ? event.getUserId().toString() : "unknown";

        CompletableFuture<SendResult<String, NotificationEvent>> future =
            kafkaTemplate.send(topic, key, event);

        future.whenComplete((result, ex) -> {
            if (ex == null) {
                log.info("✅ Published to topic: {} | userId: {} | offset: {}",
                    topic, key, result.getRecordMetadata().offset());
            } else {
                log.error("❌ Failed to publish to topic: {} | userId: {} | error: {}",
                    topic, key, ex.getMessage());
            }
        });
    }
}

5SubscriptionScheduler.java — daily job that publishes to Kafka
SubscriptionScheduler.javaNEW FILEcom.ott.subscription.scheduler
package com.ott.subscription.scheduler;

import com.ott.subscription.event.NotificationEvent;
import com.ott.subscription.model.Subscription;
import com.ott.subscription.repository.SubscriptionRepository;
import org.slf4j.*;
import org.springframework.kafka.core.KafkaTemplate;
import org.springframework.scheduling.annotation.Scheduled;
import org.springframework.stereotype.Component;
import java.time.LocalDate;
import java.time.LocalDateTime;
import java.util.List;

/**
 * PURPOSE: Daily scheduled jobs that check subscription status
 * and publish Kafka events for notification-service to act on.
 *
 * DISNEY CONTEXT: This is the scheduled background process
 * that checked for expiring/expired subscriptions and triggered
 * renewal notifications and expiry reports.
 *
 * Runs:
 * - Midnight: find subscriptions expiring in 3 days → publish to Kafka
 * - Midnight: find expired subscriptions → mark EXPIRED + publish to Kafka
 */
@Component
public class SubscriptionScheduler {

    private static final Logger log = LoggerFactory.getLogger(SubscriptionScheduler.class);

    private final SubscriptionRepository subscriptionRepository;
    private final KafkaTemplate<String, NotificationEvent> kafkaTemplate;

    public SubscriptionScheduler(
            SubscriptionRepository subscriptionRepository,
            KafkaTemplate<String, NotificationEvent> kafkaTemplate) {
        this.subscriptionRepository = subscriptionRepository;
        this.kafkaTemplate = kafkaTemplate;
    }

    /**
     * Runs every day at midnight.
     * Finds subscriptions expiring in exactly 3 days.
     * Publishes to "subscription.expiring" topic.
     * notification-service sends "renew now" reminder email.
     *
     * CRON: "0 0 0 * * *" = second=0, minute=0, hour=0, every day
     */
    @Scheduled(cron = "0 0 0 * * *")
    public void checkExpiringSubscriptions() {
        LocalDate threeDaysLater = LocalDate.now().plusDays(3);
        log.info("Scheduler: checking subscriptions expiring on {}", threeDaysLater);

        List<Subscription> expiring = subscriptionRepository
            .findByEndDateAndStatus(threeDaysLater, Subscription.SubscriptionStatus.ACTIVE);

        log.info("Found {} subscriptions expiring in 3 days", expiring.size());

        expiring.forEach(sub -> {
            NotificationEvent event = buildEvent(sub, "SUBSCRIPTION_EXPIRING");
            kafkaTemplate.send("subscription.expiring", sub.getUserId().toString(), event);
            log.info("Published subscription.expiring for userId: {}", sub.getUserId());
        });
    }

    /**
     * Runs every day at 00:05 (5 minutes after midnight).
     * Finds subscriptions whose end_date has passed.
     * Updates their status to EXPIRED in DB.
     * Publishes to "subscription.expired" topic.
     * notification-service sends "your subscription expired" email.
     */
    @Scheduled(cron = "0 5 0 * * *")
    public void processExpiredSubscriptions() {
        LocalDate today = LocalDate.now();
        log.info("Scheduler: processing subscriptions expired before {}", today);

        List<Subscription> expired = subscriptionRepository
            .findByEndDateBeforeAndStatus(today, Subscription.SubscriptionStatus.ACTIVE);

        log.info("Found {} expired subscriptions to process", expired.size());

        expired.forEach(sub -> {
            // Mark as EXPIRED in DB
            sub.setStatus(Subscription.SubscriptionStatus.EXPIRED);
            subscriptionRepository.save(sub);

            // Publish to Kafka — notification-service sends expiry email
            NotificationEvent event = buildEvent(sub, "SUBSCRIPTION_EXPIRED");
            kafkaTemplate.send("subscription.expired", sub.getUserId().toString(), event);
            log.info("Marked EXPIRED + published kafka event for userId: {}", sub.getUserId());
        });
    }

    private NotificationEvent buildEvent(Subscription sub, String eventType) {
        NotificationEvent event = new NotificationEvent();
        event.setEventType(eventType);
        event.setUserId(sub.getUserId());
        event.setPlanId(sub.getPlanId());
        event.setEventTime(LocalDateTime.now());
        // In production: call user-service Feign to get username and email
        event.setUsername("user-" + sub.getUserId());
        event.setEmail("user" + sub.getUserId() + "@espn.com");
        return event;
    }
}

2KafkaConsumerConfig.java — configures how JSON is deserialized
KafkaConsumerConfig.javaNEW FILEcom.ott.notification.config
package com.ott.notification.config;

import com.ott.notification.event.NotificationEvent;
import org.apache.kafka.clients.consumer.ConsumerConfig;
import org.apache.kafka.common.serialization.StringDeserializer;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.kafka.annotation.EnableKafka;
import org.springframework.kafka.config.ConcurrentKafkaListenerContainerFactory;
import org.springframework.kafka.core.*;
import org.springframework.kafka.support.serializer.JsonDeserializer;
import java.util.HashMap;
import java.util.Map;

/**
 * PURPOSE: Configures Kafka consumer for notification-service.
 *
 * @EnableKafka — activates @KafkaListener annotations.
 * Without this, the @KafkaListener in KafkaConsumerService
 * does nothing even if Kafka is running.
 *
 * ConcurrentKafkaListenerContainerFactory — Spring uses this
 * to create the listener containers for each @KafkaListener method.
 */
@EnableKafka
@Configuration
public class KafkaConsumerConfig {

    @Value("${spring.kafka.bootstrap-servers}")
    private String bootstrapServers;

    @Bean
    public ConsumerFactory<String, NotificationEvent> consumerFactory() {
        Map<String, Object> config = new HashMap<>();
        config.put(ConsumerConfig.BOOTSTRAP_SERVERS_CONFIG, bootstrapServers);
        config.put(ConsumerConfig.GROUP_ID_CONFIG, "notification-group");
        config.put(ConsumerConfig.AUTO_OFFSET_RESET_CONFIG, "earliest");
        config.put(ConsumerConfig.KEY_DESERIALIZER_CLASS_CONFIG, StringDeserializer.class);
        config.put(ConsumerConfig.VALUE_DESERIALIZER_CLASS_CONFIG, JsonDeserializer.class);

        // Tell JsonDeserializer which class to deserialize to
        JsonDeserializer<NotificationEvent> deserializer =
            new JsonDeserializer<>(NotificationEvent.class);
        deserializer.addTrustedPackages("*");  // trust all packages
        deserializer.setUseTypeMapperForKey(false);

        return new DefaultKafkaConsumerFactory<>(config,
            new StringDeserializer(), deserializer);
    }

    @Bean
    public ConcurrentKafkaListenerContainerFactory<String, NotificationEvent>
            kafkaListenerContainerFactory() {
        ConcurrentKafkaListenerContainerFactory<String, NotificationEvent> factory =
            new ConcurrentKafkaListenerContainerFactory<>();
        factory.setConsumerFactory(consumerFactory());
        return factory;
    }
}
